56  inline double rand_normal(double mean, double stddev) { // Box muller method
57  static double n2 = 0.0;
58  static int n2_cached = 0;
59  if (!n2_cached) {
60  double x, y, r;
61  do {
62  x = 2.0 * rand() / RAND_MAX - 1;
63  y = 2.0 * rand() / RAND_MAX - 1;
64 
65  r = x * x + y * y;
66  } while (r == 0.0 || r > 1.0);
67  {
68  double d = sqrt(-2.0 * log(r) / r);
69  double n1 = x * d;
70  n2 = y * d;
71  double result = n1 * stddev + mean;
72  n2_cached = 1;
73  return result;
74  }
75  } else {
76  n2_cached = 0;
77  return n2 * stddev + mean;
78  }
79  }
